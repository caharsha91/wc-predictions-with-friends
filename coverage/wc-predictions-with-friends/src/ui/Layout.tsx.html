function getInitials(name?: string | null, email?: string | null) {
  const base = name || email || ''
  if (!base) return 'WC'
  const parts = base.split(' ').filter(Boolean)
  if (parts.length === 1) return parts[0].slice(0, 2).toUpperCase()
  return parts
    .slice(0, 2)
    .map((part) => part[0])
    .join('')
    .toUpperCase()
}
 
function getPageTitle(pathname: string) {
  if (pathname === '/') return 'Home'
  if (pathname.startsWith('/login')) return 'Login'
  if (pathname.startsWith('/join/')) return 'Join League'
  if (pathname.startsWith('/picks/match/')) return 'Pick Match'
  if (pathname.startsWith('/picks/history')) return 'Pick History'
  if (pathname.startsWith('/picks')) return 'My Picks'
  if (pathname.startsWith('/bracket/wizard')) return 'Bracket Wizard'
  if (pathname.startsWith('/bracket/summary')) return 'Bracket Summary'
  if (pathname.startsWith('/bracket')) return 'Bracket'
  if (pathname.startsWith('/leaderboard')) return 'Leaderboard'
  if (pathname.startsWith('/matches/')) return 'Match Details'
  if (pathname.startsWith('/matches')) return 'Matches'
  if (pathname.startsWith('/rules')) return 'Rules'
  if (pathname.startsWith('/settings')) return 'Settings'
  if (pathname === '/admin') return 'Admin'
  if (pathname.startsWith('/admin/locks')) return 'Admin Locks'
  if (pathname.startsWith('/admin/sync')) return 'Admin Sync'
  if (pathname.startsWith('/admin/league')) return 'Admin League'
  return 'World Cup Predictions'
}
